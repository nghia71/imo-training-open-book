\begin{problem}[\gls{MEMO 2015}/I/P4]\label{problem:MEMO-2015-I-P4}
    Tìm tất cả các cặp số nguyên dương \( (m, n) \) sao cho tồn tại các số nguyên \( a, b > 1 \) nguyên tố cùng nhau, thỏa mãn:
    \[
        \frac{a^m + b^m}{a^n + b^n} \in \mathbb{Z}.
    \]
\end{problem}
