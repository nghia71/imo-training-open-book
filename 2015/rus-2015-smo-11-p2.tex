\begin{problem}[\gls{RUS 2015 SMO}/11/P2]\label{problem:RUS-2015-SMO-11-P2}
    Cho \( a, b > 1 \) là các số tự nhiên, và \( a^2 + b,\ a + b^2 \) đều là các số nguyên tố.  
    Chứng minh rằng:
    \[
        \gcd(ab + 1,\ a + b) = 1.
    \]
\end{problem}
