\begin{problem}[\gls{RUS 2015 MO}/9/P3]\label{example:RUS-2015-MO-9-P3}
    Cho \( a, x, y \) là các số nguyên dương thỏa mãn \( a > 100,\ x > 100,\ y > 100 \) và
    \[
        y^2 - 1 = a^2(x^2 - 1).
    \]
    Tìm giá trị nhỏ nhất của \( \frac{a}{x} \).    
\end{problem}
