\begin{problem}[\gls{BW 2015}/P16]\label{example:BW-2015-P16}
    Ký hiệu \( P(n) \) là ước số nguyên tố lớn nhất của \( n \).  
    Tìm tất cả các số nguyên \( n \geq 2 \) sao cho:
    \[
        P(n) + \lfloor \sqrt{n} \rfloor = P(n+1) + \lfloor \sqrt{n+1} \rfloor.
    \]
\end{problem}
