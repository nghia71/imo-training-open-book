\begin{theorem*}[GCD Power Sum Identity]
    \label{theorem:gcd-power-sum}
    For any positive integer \( n \) and any real or complex number \( a \), we have
    \[
        \sum_{k=1}^n a^{\gcd(k,n)} = \sum_{d \mid n} \phi(d) \, a^{n/d}.
    \]
\end{theorem*}
