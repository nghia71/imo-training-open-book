\begin{exercise*}[\gls{RUS 2015 TST}/D10/P3]\label{example:RUS-2015-TST-D10-P3}[\textbf{unrated}]
	Tìm tất cả các số nguyên \( k \) sao cho tồn tại vô số bộ ba số nguyên \( (a, b, c) \) thỏa mãn:
	\[
		(a^2 - k)(b^2 - k) = c^2 - k.
	\]
\end{exercise*}
